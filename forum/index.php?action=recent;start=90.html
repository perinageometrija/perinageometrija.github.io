Kod Izaberi Expand

program randomxybrojeva;

{$APPTYPE CONSOLE}

uses
  SysUtils;

var
    f:text;
    i:integer;
    g:integer;
    s:string;
begin
writeln('Napisite ime fajla:');
read(s);
assign(f,s+'.txt');
rewrite(f);
Randomize();
i:=1;
while i<=1000 do
begin
g:=random(20000) - 10000;
writeln(f,g);
i:=i+1;
end;
writeln('Ispisao!!!');
readln;
readln;
end.
